# 简单排序算法

## 选择排序

```cpp
for (int i = 0; i < n - 1; i++){
    for (int j = i + 1; j < n; j++){
        if (a[i] < a[j]){
            swap(a[i],a[j]);
        }
    }
}



void selectionSort(int arr[], int n) {
    for (int i = 0; i < n - 1; ++i) {
        int minIndex = i; // 记录最小元素的索引
        // 在未排序部分找到最小元素的索引
        for (int j = i + 1; j < n; ++j) {
            if (arr[j] < arr[minIndex]) {
                minIndex = j;
            }
        }
        // 将最小元素与未排序部分的第一个元素交换位置
        swap(a[i],a[minIndex]);
    }
}
```

## 冒泡排序
```cpp
for (int i = 0; i < n ; i++) {
    for (int j = 0; j < n - i - 1; j++) {
        if (a[j] > a[j + 1]) {
            swap(a[j], a[j + 1]);
        }
    }
}



void bubbleSort(int arr[], int n) {
    bool swapped; // 用于记录是否发生了交换
    for (int i = 0; i < n - 1; ++i) {
        swapped = false; // 初始化为 false
        // 每次遍历将最大的元素交换到末尾
        for (int j = 0; j < n - i - 1; ++j) {
            // 如果当前元素大于下一个元素，则交换它们
            if (arr[j] > arr[j + 1]) {
                swap(a[j],a[j + 1]);
                swapped = true; // 设置为 true
            }
        }
        // 如果在本次遍历中没有发生交换，则表示列表已经排序完成，提前结束循环
        if (!swapped) {
            break;
        }
    }
}


```

## 插入排序
```cpp
for (int i = 1; i < n; ++i) {
    int t = a[i];
    int j = i - 1;
    while (j >= 0 && a[j] > t) {
        a[j + 1] = a[j];
        j = j - 1;
    }
    a[j + 1] = t;
}
```


## 快速排序
```cpp

void quick_sort(int l,int r){
    if(l >= r) return;
    
    int i = l - 1,j = r + 1 ,x = q[(l + r) >> 1];
    while (i < j)
    {
        do i ++ ; while (q[i] < x);
        do j -- ; while (q[j] > x);
        if (i < j) swap(q[i], q[j]);
    }
    
    quick_sort(l , j);
    quick_sort(j + 1, r);
    
}

```

## 归并排序

void merge_sort(int q[], int l, int r)
{
    if (l >= r) return;

    int mid = l + r >> 1;

    merge_sort(q, l, mid), merge_sort(q, mid + 1, r);

    int k = 0, i = l, j = mid + 1;
    while (i <= mid && j <= r)
        if (q[i] <= q[j]) tmp[k ++ ] = q[i ++ ];
        else tmp[k ++ ] = q[j ++ ];
    while (i <= mid) tmp[k ++ ] = q[i ++ ];
    while (j <= r) tmp[k ++ ] = q[j ++ ];

    for (i = l, j = 0; i <= r; i ++, j ++ ) q[i] = tmp[j];
}
